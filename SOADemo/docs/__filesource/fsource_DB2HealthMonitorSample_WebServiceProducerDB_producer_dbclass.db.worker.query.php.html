  70.     /**
  71.      * Executes the desired SQL query statement.
  72.      * @access public
  73.      * @param resource $connection the connection to the DB2 database
  74.      * @param string the SQL statement to be prepared and executed
  75.      * @return mixed (array of row results on success; {@link DBError} object on failure)
  76.      */
  77.     public static function DBQuery($connection, $sql)
  78.     {
  79.         //Prepare the desired SQL statement
  80.         $stmt = db2_prepare($connection, $sql);
  81.         $rows = array();
  82.         
  83.         if($stmt)
  84.         {
  85.             //Execute the desired SQL statement
  86.             $result = db2_execute($stmt);
  87.             
  88.             //Fetch the Result Set Rows
  89.             if($result)
  90.             {
  91.                 while($row = db2_fetch_array($stmt))
  92.                     $rows[] = $row;    
  93.             }
  94.             else 
  95.             {
  96.                 return new DBError(DBError::DB_QUERY_EXECUTION_ERROR_CD, DBError::DB_QUERY_EXECUTION_ERROR_MSG,
  97.                                    db2_stmt_error(), db2_stmt_errormsg());
  98.             }
  99.         }    
  100.         else 
  101.         {
  102.             return new DBError(DBError::DB_QUERY_PREPARE_ERROR_CD, DBError::DB_QUERY_PREPARE_ERROR_MSG,
  103.                                    db2_stmt_error(), db2_stmt_errormsg());
  104.         }
  105.         
  106.         //Free the statement and result memory
  107.         self::DBFree($stmt);
  108.         
  109.         return $rows;
  110.     }

  204.     /**
  205.      * Prepares and executes the GET_DBSIZE_INFO stored procedure (required prior to executing the desired XQUERY).
  206.      * The information in the XQUERYs of this application rely on a cached value to be present, hence, this stored
  207.      * procedure being executed guarantees that the subsequent queries will not fail.
  208.      * @access public
  209.      * @param resource $connection the connection to the specified DB2 database
  210.      * @return mixed (true on success; {@link DBError} object on failure)
  211.      */
  212.     public static function DBQuerySize($connection)
  213.     {
  214.         //Prepares the Stored Procedure Call
  215.         $stmt = db2_prepare($connection, 'CALL GET_DBSIZE_INFO(?,?,?,-1)');
  216.         
  217.         //Initializes the Output Parameters with the correct size/type required by the procedure
  218.         $timestamp = str_pad('',50);
  219.         $size = 0.0;
  220.         $capacity = 0.0;
  221.         
  222.         if($stmt)
  223.         {    
  224.             //Bind all Output Parameters
  225.             db2_bind_param($stmt, 1, 'timestamp', DB2_PARAM_OUT);
  226.             db2_bind_param($stmt, 2, 'size', DB2_PARAM_OUT);
  227.             db2_bind_param($stmt, 3, 'capacity', DB2_PARAM_OUT);
  228.             
  229.             //Execute the stored procedure call
  230.             if(!db2_execute($stmt))
  231.             {
  232.                 return new DBError(DBError::DB_QUERY_EXECUTION_ERROR_CD, DBError::DB_QUERY_EXECUTION_ERROR_MSG,
  233.                                    db2_stmt_error(), db2_stmt_errormsg());                
  234.             }
  235.         }
  236.         else 
  237.         {
  238.             return new DBError(DBError::DB_QUERY_PREPARE_ERROR_CD, DBError::DB_QUERY_PREPARE_ERROR_MSG,
  239.                                    db2_stmt_error(), db2_stmt_errormsg());            
  240.         }    
  241.         
  242.         return true;
  243.     }
