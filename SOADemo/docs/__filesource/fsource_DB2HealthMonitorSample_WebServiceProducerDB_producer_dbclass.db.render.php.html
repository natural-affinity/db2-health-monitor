  83.     /**

  89.     /** 
  90.      * Validates all Web Service Function input, and developer inputs.
  91.      * @access private
  92.      * @param string $user username used to access the database
  93.      * @param string $pass password that corresponds with the above username
  94.      * @param string $db the name of the database to connect to

  103.     private static function DBValidate($user, $pass, $db, $port, $def, $det, $his, $lang, $type)
  104.     {    
  105.         //Database Username Validation
  106.         if(!preg_match(self::REGEX_DATABASE_PARAMETER, $user))
  107.             return new DBError(DBError::DB_USERNAME_PARAM_ERROR_CD, DBError::DB_USERNAME_PARAM_ERROR_MSG, null, null);
  108.         
  109.         //Database Password Validation
  110.         if(!preg_match(self::REGEX_DATABASE_PARAMETER, $pass))
  111.             return new DBError(DBError::DB_PASSWORD_PARAM_ERROR_CD, DBError::DB_PASSWORD_PARAM_ERROR_MSG, null, null);
  112.         
  113.         //Database Name Validation
  114.         if(!preg_match(self::REGEX_DATABASE_PARAMETER, $db))
  115.             return new DBError(DBError::DB_DATABASE_PARAM_ERROR_CD, DBError::DB_DATABASE_PARAM_ERROR_MSG, null, null);
  116.         
  117.         //Database Port Validation
  118.         if(!preg_match(self::REGEX_PORT_PARAMETER, $port))
  119.             return new DBError(DBError::DB_PORT_PARAM_ERROR_CD, DBError::DB_PORT_PARAM_ERROR_MSG, null, null);
  120.         
  121.         //Database Language Code Validation
  122.         if(!preg_match(self::REGEX_LANGUAGE_CODE, $lang))
  123.             return new DBError(DBError::DB_LANGUAGE_PARAM_ERROR_CD, DBError::DB_LANGUAGE_PARAM_ERROR_MSG, null, null);
  124.         
  125.         //Object Type Parameter Validation
  126.         if(!preg_match(self::REGEX_OBJECT_TYPE, $type))
  127.             return new DBError(DBError::DB_OBJTYPE_PARAM_ERROR_CD, DBError::DB_OBJTYPE_PARAM_ERROR_MSG, null, null);
  128.         
  129.         //Definition Flag Validation
  130.         if(gettype($def) != 'boolean')
  131.             return new DBError(DBError::DB_DEFFLAG_PARAM_ERROR_CD,DBError::DB_DEFFLAG_PARAM_ERROR_MSG, null, null);
  132.         
  133.         //Details Flag Validation
  134.         if(gettype($det) != 'boolean')
  135.             return new DBError(DBError::DB_DETFLAG_PARAM_ERROR_CD, DBError::DB_DETFLAG_PARAM_ERROR_MSG, null, null);
  136.         
  137.         //History Flag Validation
  138.         if(gettype($his) != 'boolean')
  139.             return new DBError(DBError::DB_HISFLAG_PARAM_ERROR_CD, DBError::DB_HISFLAG_PARAM_ERROR_MSG, null, null);
  140.             
  141.         return true;
  142.     }

  144.     /**
  145.      * Utilizes the {@link DBConnectionWorker} and the {@link DBQueryWorker} classes, along with input validation
  146.      * mechanisms, to obtain an XML representation of the DB2 Health Report.
  147.      * @access public
  148.      * @param string $user username used to access the database
  149.      * @param string $pass password that corresponds with the above username
  150.      * @param string $db the name of the database to connect to
  151.      * @param string $port the port number under which the database resides
  152.      * @param boolean $def the definition request flag used to build the query
  153.      * @param boolean $det the details request flag used to build the query
  154.      * @param boolean $his the history request flag used to build the query
  155.      * @param string $lang the language code used to build the query
  156.      * @param string $type the type flag used to select the appropriate query and parameters
  157.      * @return string 
  158.      */
  159.     public static function DBReport($user, $pass, $db, $port, $def, $det, $his, $lang, $type)
  160.     {    
  161.         $xml = '';
  162.         $sql = '';
  163.         $connection = null;
  164.         
  165.         //Validate all input parameters
  166.         $result = self::DBValidate($user, $pass, $db, $port, $def, $det, $his, $lang, $type);
  167.         
  168.         //Build the appropriate query and establish a database connection
  169.         if(!($result instanceof DBError))
  170.         {
  171.             $sql = ($type == 'all') 
  172.                         ? DBQueryWorker::DBXQueryAssembleAll($db, $lang, $def, $det, $his)
  173.                         : DBQueryWorker::DBXQueryAssembleType($db, $lang, $def, $det, $his, $type);    
  174.                         
  175.             $result = DBConnectionWorker::DBConnect($user, $pass, $db, $port);
  176.         }
  177.             
  178.         //Execute a DB2 Stored Procedure to prepare the data for the XQUERY to follow
  179.         if(!($result instanceof DBError))
  180.         {
  181.             $connection = $result;
  182.             $result = DBQueryWorker::DBQuerySize($connection);    
  183.         }    
  184.  
  185.         //Prepare and Execute the desired Query
  186.         if(!($result instanceof DBError))
  187.         {
  188.             $result = DBQueryWorker::DBXQuery($connection, $sql);
  189.         }
  190.         
  191.         //Store the Result if the Query was successful
  192.         if(!($result instanceof DBError))
  193.         {
  194.             $xml = $result;
  195.         }
  196.         else 
  197.         {
  198.             $xml = $result->__toString();
  199.             $result->logError($xml);            
  200.         }
  201.         
  202.         //Close the Database Connection
  203.         DBConnectionWorker::DBClose($connection);
  204.         
  205.         return $xml;
  206.     }
